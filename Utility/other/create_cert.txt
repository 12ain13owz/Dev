1. Create Private Key - openssl genrsa -out ssl_private.key 4096

2. Create Pubilc Key - openssl rsa -in ssl_private.key -pubout -out ssl_public.key

3. init ssl.cnf
[req]
default_bits = 4096
prompt = no
default_md = sha256
x509_extensions = v3_req
distinguished_name = dn
[dn]
C = TH
ST = Bangkok
L = Bangkok
O = None
OU = None
emailAddress = localhost@example.com
CN = localhost
[v3_req]
subjectAltName = @alt_names
[alt_names]
DNS.1 = localhost
IP.1 = 192.168.1.46

4. Create CSR - openssl req -new -key ssl_private.key -out ssl.csr -config ssl.cnf

5. Create CRT - openssl x509 -req -in ssl.csr -signkey ssl_private.key -out ssl.crt -extensions v3_req -extfile ssl.cnf
